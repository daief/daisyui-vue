Controlled swap

PreviewCode
 / 
import { ref } from 'vue';

export default {
  setup: () => {
    const active = ref(false);
    return () => (
      <div class="text-center">
        <div class="py-2">
          <dv-button onClick={() => (active.value = true)}>on</dv-button>
          &nbsp;/&nbsp;
          <dv-button onClick={() => (active.value = false)}>off</dv-button>
        </div>
        <div class="text-9xl">
          <dv-swap on="🥳" off="😭" animation="flip" active={active.value} />
        </div>
      </div>
    );
  },
};
